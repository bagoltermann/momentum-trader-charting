# INSTALLATION:
#   cd momentum-trader-charting/backend
#   python -m venv venv
#   venv\Scripts\activate  # On Windows
#   source venv/bin/activate  # On Mac/Linux
#   pip install -r requirements.txt

#   - Momentum Trader app must be set up with valid Schwab tokens
#   - Tokens are shared between apps (per-machine, not per-app)
